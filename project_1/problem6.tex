\section{Overview}
We solve a joint \emph{capacity expansion and economic dispatch} problem
on a two-node power system with three generators connected by a single
transmission line. The objective is to choose installed generator
capacities $x \in \mathbb{R}^{n_g}_{\ge 0}$ that minimize the sum of
annualized investment (CAPEX) costs and operational (OPEX) costs over
$T$ hourly dispatch periods.

We solve the problem three ways and compare the results:
\begin{enumerate}
    \item A direct LP solve (HiGHS) as a reference benchmark.
    \item \textbf{Kelley's cutting-plane method} applied to the outer
          (capacity) problem, with the inner dispatch LP acting as an
          oracle returning function values and subgradients.
    \item A \textbf{projected subgradient method} on the same outer
          problem.
\end{enumerate}

\section{System Data}
\begin{itemize}
    \item Number of generators: $n_g = 3$.
    \item Number of nodes: $n_b = 2$, with generators 1 and 2 at node 1
          and generator 3 at node 2.
    \item Number of transmission lines: $n_\ell = 1$, connecting node 1 to
          node 2 with incidence vector $A = (1,\, -1)^\top$ and flow limit
          $f^{\max} = 80$ MW.
    \item Dispatch horizon: $T = 600$ hourly steps, generated by tiling a
          24-hour base demand shape for $n_b$ demand nodes with small
          i.i.d.\ day-scale noise ($\sigma = 0.05$).
    \item OPEX costs: $c^{\text{op}} = (30,\, 20,\, 40)$ \$/MWh.
    \item CAPEX costs: $c^{\text{cap}} = (100,\, 200,\, 80)$ \$/MW.
    \item Capacity bounds: $0 \le x_i \le 10^4$ MW.
\end{itemize}

\section{Full Monolithic LP (Reference)}
Let $g_{i,t}$ denote dispatch of generator $i$ at time $t$,
$f_{\ell,t}$ the flow on line $\ell$, and $x_i$ the installed capacity
of generator $i$. Let $d_{n,t}$ denote the demand at node $n$ at
time $t$.
\begin{align}
    \min_{x,\, g,\, f}\quad
        & \sum_{i=1}^{n_g}\sum_{t=1}^{T} c^{\text{op}}_i\, g_{i,t}
        + \sum_{i=1}^{n_g} c^{\text{cap}}_i\, x_i \label{eq:full-obj} \\
    \text{s.t.}\quad
        & g_{i,t} \le x_i, && \forall\, i,\, t
          \label{eq:cap} \\
        & g_{1,t} + g_{2,t} + f_{1,t} = d_{1,t}, && \forall\, t
          \label{eq:bal-n1} \\
        & g_{3,t} - f_{1,t} = d_{2,t}, && \forall\, t
          \label{eq:bal-n2} \\
        & -f^{\max} \le f_{\ell,t} \le f^{\max}, && \forall\, \ell,\, t \\
        & 0 \le g_{i,t}, \quad 0 \le x_i \le \bar{x}.
\end{align}
This LP is solved directly with HiGHS and provides the benchmark
optimal value $f^\star$ and capacities $x^\star$.

\section{Benders-Style Decomposition}

\subsection{Outer problem and dispatch oracle}
We decompose the problem into an \emph{outer} problem over the capacity
variable $x$ and an \emph{inner} dispatch subproblem $Q(x)$:
\begin{equation}
    \min_{x \in X}\quad F(x) \;:=\; c^{\text{cap}\,\top} x + Q(x),
    \qquad X = \{x \in \mathbb{R}^{n_g}:\ 0 \le x \le \bar{x}\}.
\end{equation}
The inner dispatch value $Q(x)$ is defined by the parameterized LP
\begin{align}
    Q(x) \;=\; \min_{g,\, f,\, s}\quad
        & \sum_{i,t} c^{\text{op}}_i\, g_{i,t}
        + \text{VOLL} \sum_{n,t} s_{n,t} \\
    \text{s.t.}\quad
        & g_{i,t} \le x_i, && \forall\, i,\, t \quad (\lambda_{i,t}) \\
        & g_{1,t} + g_{2,t} + f_{1,t} + s_{1,t} = d_{1,t}, && \forall\, t \\
        & g_{3,t} - f_{1,t} + s_{2,t} = d_{2,t}, && \forall\, t \\
        & -f^{\max} \le f_{\ell,t} \le f^{\max},\quad g_{i,t}\ge 0,\quad s_{n,t}\ge 0.
\end{align}
The load-shedding slack $s_{n,t}$ penalized at
$\text{VOLL} = 10^6$ keeps the oracle feasible for every $x \ge 0$, so
$F(x)$ is finite and convex on $X$ as a value function of a convex LP
parameterized linearly in its right-hand side.

\subsection{Subgradient via LP duality (envelope theorem)}
Let $\lambda_{i,t}(x) \le 0$ be the optimal dual of the capacity
constraint $g_{i,t} \le x_i$. By LP duality / the envelope theorem,
\begin{equation}
    \frac{\partial Q}{\partial x_i}
    \;=\; \sum_{t=1}^{T} \lambda_{i,t}(x),
    \qquad
    \nabla F(x)_i
    \;=\; c^{\text{cap}}_i + \sum_{t=1}^{T} \lambda_{i,t}(x),
\end{equation}
which is the (sub)gradient supplied by the oracle \texttt{functionAndGradient}.

\subsection{Kelley's cutting-plane method}
$F$ is convex, so every $(F(x_k),\, g_k)$ produced by the oracle
yields a valid linear lower bound
$F(x) \ge F(x_k) + g_k^\top (x - x_k)$. Kelley's method maintains the
piecewise-linear lower envelope
\begin{equation}
    \hat F_K(x) \;=\; \max_{k \le K}\, \bigl\{ F(x_k) + g_k^\top (x - x_k) \bigr\},
\end{equation}
and at iteration $K$ solves the master LP
\begin{equation}
    x_{K+1} \in \arg\min_{x \in X,\, \theta} \;
        \{\, \theta \,:\, \theta \ge F(x_k) + g_k^\top (x - x_k),\ k = 1,\dots,K \,\}.
\end{equation}
The master value $\theta_{K+1}$ is a valid lower bound (LB), and the
oracle value $F(x_{K+1})$ is an upper bound (UB). The algorithm stops
when the relative gap $(\text{UB} - \text{LB}) / |\text{UB}|$ is below
\texttt{OPT\_TOL} $= 10^{-5}$.

\subsection{Projected subgradient method}
The subgradient method generates iterates
\begin{equation}
    x_{k+1} \;=\; \Pi_X\bigl( x_k - \alpha_k\, g_k \bigr),
\end{equation}
where $g_k = \nabla F(x_k)$ is returned by the same dispatch oracle,
$\Pi_X$ is the Euclidean projection onto the box $X$, and $\alpha_k$
is a diminishing step size. Unlike Kelley's method, this scheme does
not maintain an explicit lower-bounding model of $F$.

\section{Reported Quantities and Comparison}
For each method we record:
\begin{itemize}
    \item Final objective $F(x)$ split into CAPEX
          $c^{\text{cap}\,\top} x$ and OPEX $Q(x)$.
    \item Final optimality gap, iteration count, and CPU time.
    \item Installed capacities $x^\star$ (written to
          \texttt{capacities\_*.csv}).
    \item Hourly dispatch $g_{i,t}$ and flows $f_{\ell,t}$ recovered by
          re-solving the dispatch oracle at $x^\star$.
\end{itemize}
The comparison artifacts in
\texttt{results/problem6compare/} overlay HiGHS, Kelley, and
subgradient capacities and dispatch schedules for visual inspection.
